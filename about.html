Code Example

Kotlin
fun main() {
    // Welcome to Chik Chok IDE
    val message = "Hello Developer!"
    
    // Calculate Fibonacci sequence
    fun fib(n: Int): Int = when (n) {
        0, 1 -> n
        else -> fib(n - 1) + fib(n - 2)
    }
    
    println("$message Fibonacci of 5 is ${fib(5)}")
}
JavaScript
// Async data fetching example
async function fetchData() {
    try {
        const response = await fetch('/api/data');
        const data = await response.json();
        
        // Process data with Chik Chok IDE
        return data.map(item => ({
            ...item,
            processed: true
        }));
    } catch (error) {
        console.error('Fetch error:', error);
        throw error;
    }
}
